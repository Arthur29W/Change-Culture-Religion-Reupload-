ccr_small_change = {
	if = {
		limit = { has_game_rule = shorter_ccr_mod_length }
		add_modifier = {
			name = ccr_recent_changes
			is_decaying = yes
			years = 1
		}
	}
	else_if = {
		limit = { has_game_rule = default_ccr_mod_length }
		add_modifier = {
			name = ccr_recent_changes
			is_decaying = yes
			years = 3
		}
	}
	else_if = {
		limit = { has_game_rule = longer_ccr_mod_length }
		add_modifier = {
			name = ccr_recent_changes
			is_decaying = yes
			years = 5
		}
	}
}

ccrr_update_conversion_perc = {
	set_variable = {
		name = ccr_conversion_perc
		value = 0.004
	}
		
	if = {
		limit = { has_law = law_type:law_religious_schools }
		if = {
			limit = {
				institution_investment_level = {
					institution = institution_schools
					value = 5
				}
			}
			change_variable = {
				name = ccr_conversion_perc
				multiply = 2
			}
		} else_if = {
			limit = {
				institution_investment_level = {
					institution = institution_schools
					value = 4
				}
			}
			change_variable = {
				name = ccr_conversion_perc
				multiply = 1.8
			}
		} else_if = {
			limit = {
				institution_investment_level = {
					institution = institution_schools
					value = 3
				}
			}
			change_variable = {
				name = ccr_conversion_perc
				multiply = 1.6
			}
		} else_if = {
			limit = {
				institution_investment_level = {
					institution = institution_schools
					value = 2
				}
			}
			change_variable = {
				name = ccr_conversion_perc
				multiply = 1.4
			}
		} else = {
			change_variable = {
				name = ccr_conversion_perc
				multiply = 1.2
			}
		}
	}
}

ccrr_ig_devout_set_piousness = {
	if = {
		limit = {
			ig:ig_devout = { has_ideology = ideology_pious }
			country_has_state_religion = rel:mahayana
			OR = {
				ccrr_tag = { TAG = "CHI" }
				ccrr_tag = { TAG = "KOR" }
			}
		}
		ig:ig_devout = { remove_ideology = ideology_pious }
	} else_if = {
		limit = { ig:ig_devout = { NOT = { has_ideology = ideology_pious } } }
		add_ideology = ideology_pious
		ig:ig_devout = { add_ideology = ideology_pious }
	}
}

ccrr_ig_devout_set_morality = {
	if = {
		limit = {
			OR = {
				country_has_state_religion = rel:gelugpa
				country_has_state_religion = rel:theravada
			}
		}
		ig:ig_devout = { add_ideology = ideology_buddhist_moralist }
	} else_if = {
		limit = { country_has_state_religion = rel:hindu }
		ig:ig_devout = { add_ideology = ideology_hindu_moralist }
	} else_if = {
		limit = { country_has_state_religion = rel:mahayana }
		if = {
			limit = {
				OR = {
					ccrr_tag = { TAG = "CHI" }
					ccrr_tag = { TAG = "KOR" }
				}
			}
			ig:ig_devout = { add_ideology = ideology_confucian }
		} else = {
			ig:ig_devout = { add_ideology = ideology_buddhist_moralist }
		}
	} else_if = {
		limit = {
			country_has_state_religion = rel:protestant
			ccrr_tag = { TAG = "TPG" }
		}
		ig:ig_devout = { add_ideology = ideology_heavenly_kingdom_theocratic }
	} else_if = {
		limit = { country_has_state_religion = rel:shinto }
		ig:ig_devout = { add_ideology = ideology_shinto_moralist }
	} else_if = {
		limit = { country_has_state_religion = rel:sikh }
		ig:ig_devout = { add_ideology = ideology_sikh_moralist }
	} else = {
		ig:ig_devout = { add_ideology = ideology_moralist }
	}
}

ccrr_ig_devout_set_patriarchy = {
	if = {
		limit = { country_has_state_religion = rel:oriental_orthodox }
		ig:ig_devout = { add_ideology = ideology_oriental_orthodox_patriarch }
	} else_if = {
		limit = { country_has_state_religion = rel:orthodox }
		if = {
			limit = { 
				exists = c:RUS
				OR = {
					c:RUS = ROOT
					AND = {
						c:RUS = { country_has_state_religion = rel:orthodox }
						is_subject_of = c:RUS
					}
				}
			}
			ig:ig_devout = { add_ideology = ideology_russian_patriarch }
		} else = {
			ig:ig_devout = { add_ideology = ideology_orthodox_patriarch }
		}
	} else_if = {
		limit = { has_law = law_type:law_womens_suffrage }
		ig:ig_devout = { add_ideology = ideology_patriarchal_suffrage }
	} else = {
		ig:ig_devout = { add_ideology = ideology_patriarchal }
	}
}

ccrr_update_characters = {
	every_scope_character = {
		limit = { has_variable = ccr_character_modified }
		remove_variable = ccr_character_modified
	}
	
	hidden_effect = {
		random_primary_culture = { save_scope_as = ccr_newchar_culture }
		
		ig:ig_devout = {
			leader = {
				kill_character = {
					hidden = yes
					value = yes
				}
			}
		}
		
		create_character = {
			ig_leader = yes
			interest_group = ig_devout
			culture = scope:ccr_newchar_culture
			religion = ROOT.var:ccr_new_religion
			historical = no
			female = no
		}
	}
}
